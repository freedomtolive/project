		<script type="text/jsx">
			var ClickApp = React.createClass({
				getInitialState:function(){
					return {
						clickCount:0
					}
				},
				clickMe:function(){
					this.setState({
						clickCount:this.state.clickCount + 1
					})
				},
				render:function(){
					return (
						<div>
							<h2>请点击下面的按钮</h2>
							<button onClick={this.clickMe}>点击我增加数字吧</button>
							<div >
								共点击了{this.state.clickCount}次
							</div>
						</div>
					)
				}
			})

			var clickComponent = React.render(
				<ClickApp/>,
				document.getElementById("app")
			)
		</script>
